class allayAll{
	private $showArray;
	private $rowSize;
	private $lineSize;
	function __construct($rowSize, $lineSize){
		$this->rowSize = $rowSize;
		$this->lineSize = $lineSize;
	}
	function setArray($inputArray){
		$this->showArray = $inputArray;
	}
	function showArray(){
		$showString = '';
		for($i=0; $i < $this->rowSize; $i ++){
			for($j=0; $j < $this->lineSize; $j ++){
				$showString .= $this->showArray[$i][$j];
			}
			$showString .= "\n";
		}
		echo($showString);
	}
	//テスト用配列
	function makeTestArray(){
		$theArray = array();
		for($i=0; $i < $this->rowSize; $i ++){
			for($j=0; $j < $this->lineSize; $j ++){
				$theArray[$i][$j] = $i + 2 * $j;
			}
		}
		return $theArray;
	}
}
class checkLine{
	private $problemArray;
	private $convertedArray;
	private $answerArray;
	function __construct($problemArray){
		$this->problemArray = $problemArray;
		$this->convertedArray = $this->convertArray();
		$this->answerArray = $this->makeAnswerArray(10, 10);
	}
	function setlowest(){
		$arrayLength = count($this->answerArray);
		$checkingPoint = 0;
		$comparingValue = 0;
		for($nowPoint = 0; $nowPoint < $arrayLength; $nowPoint++){
			if($nowPoint >= $comparingValue && $checkingPoint < count($this->convertedArray)){
				$comparingValue += $this->convertedArray[$checkingPoint];
				$checkingPoint += 1;
			}
			$this->answerArray[$nowPoint]['lowest'] = $checkingPoint;
			var_export($this->answerArray[$nowPoint]['lowest']);
			echo("checking" . $checkingPoint . " sum" . $comparingValue . " now " . $nowPoint . "\n");
		}
	}
	function setHighest($convertedArray){
		$arrayLength = count($this->answerArray);
		$checkingPoint = count($this->convertedArray);
		$comparingValue = $this->convertedArray[$checkingPoint];
		for($nowPoint = 0; $nowPoint < $arrayLength; $nowPoint++){
			if($nowPoint >= $comparingValue && $checkingPoint > 0){
				$checkingPoint -= 1;
				$comparingValue += $this->convertedArray[$checkingPoint];
			}
			$this->answerArray[$arrayLength - $nowPoint]['lowest'] = $checkingPoint;
			var_export($this->answerArray[$arrayLength - $nowPoint]['lowest']);
			echo("checkinghigh" . $checkingPoint . " sum" . $comparingValue . " now " . ($arrayLength - $nowPoint) . "\n");
		}
	}
	function convertArray(){
		return array(2,1,5);
	}
	function makeAnswerArray($rowSize, $lineSize){
		$theArray = array();
		for($i=0; $i < $rowSize; $i ++){
			for($j=0; $j < $lineSize; $j ++){
				$theArray[$i][$j] = array();
			}
		}
		return $theArray;
	}
}
